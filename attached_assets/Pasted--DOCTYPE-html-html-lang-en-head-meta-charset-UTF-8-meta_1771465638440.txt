    <script>
        // Data
        const TARGET_TEXT = "Architecture Ideas:\n\n1. Use modular components for the UI.\n2. Implement dark mode by default.\n3. Ensure touch targets are >44px.\n\nâ€” Needs review by Friday.";
        
        // DOM Elements
        const scanBtn = document.getElementById('scan-btn');
        const shutterInner = document.getElementById('shutter-inner');
        const scannerBeam = document.getElementById('scanner-beam');
        const sourceImage = document.getElementById('source-image');
        const detectionLayer = document.getElementById('detection-layer');
        const outputText = document.getElementById('output-text');
        const cursor = document.getElementById('cursor');
        const scanPrompt = document.getElementById('scan-prompt');
        const statusPill = document.getElementById('status-pill');
        const statusText = document.getElementById('status-text');
        const resultSheet = document.getElementById('result-sheet');
        const closeResult = document.getElementById('close-result');

        let isScanning = false;

        // Generate visual detection boxes
        function createDetectionBox(yPos) {
            const box = document.createElement('div');
            const width = Math.floor(Math.random() * 30) + 20;
            const left = Math.floor(Math.random() * 40) + 15;
            
            box.className = 'absolute border border-blue-400/80 bg-blue-400/20 rounded z-10';
            box.style.left = `${left}%`;
            box.style.top = `${yPos}%`;
            box.style.width = `${width}%`;
            box.style.height = '20px';
            box.style.opacity = '0';
            
            detectionLayer.appendChild(box);

            gsap.to(box, {
                opacity: 1,
                scale: 1,
                duration: 0.1,
                onComplete: () => {
                    gsap.to(box, {
                        opacity: 0,
                        delay: 0.3,
                        duration: 0.2,
                        onComplete: () => box.remove()
                    });
                }
            });
        }

        function typeWriter(text, element, speed = 30) {
            let i = 0;
            element.innerHTML = '';
            cursor.classList.remove('hidden');
            
            function type() {
                if (i < text.length) {
                    const randomSpeed = speed + (Math.random() * 40 - 20);
                    if (text.charAt(i) === '\n') {
                        element.innerHTML += '<br/>';
                    } else {
                        element.innerHTML += text.charAt(i);
                    }
                    i++;
                    // Auto scroll to bottom of text
                    element.parentElement.scrollTop = element.parentElement.scrollHeight;
                    setTimeout(type, randomSpeed);
                } else {
                    cursor.classList.add('hidden');
                }
            }
            type();
        }

        // Close Result Sheet
        closeResult.addEventListener('click', () => {
            resultSheet.classList.add('translate-y-[110%]');
            resultSheet.classList.remove('translate-y-0');
            
            // Reset UI State after delay
            setTimeout(() => {
                isScanning = false;
                scanPrompt.style.opacity = '1';
                shutterInner.classList.remove('scale-[0.4]', 'bg-red-500');
                sourceImage.style.filter = 'grayscale(100%) opacity(0.6)';
                sourceImage.style.transform = 'scale(1)';
            }, 300);
        });

        // Main Scan Action
        scanBtn.addEventListener('click', () => {
            if (isScanning) return;
            isScanning = true;
            
            // 1. Shutter Feedback
            shutterInner.classList.add('scale-[0.4]', 'bg-red-500');
            
            // 2. Hide Prompt / Show Image
            scanPrompt.style.opacity = '0';
            gsap.to(sourceImage, { filter: 'grayscale(0%) opacity(1)', scale: 1.1, duration: 0.5 });
            
            // 3. Show Status
            statusPill.classList.remove('opacity-0', 'translate-y-4');
            statusText.innerText = 'Scanning...';

            // 4. Scanner Beam Animation
            scannerBeam.classList.remove('hidden');
            const tl = gsap.timeline();

            // Downward Scan
            tl.fromTo(scannerBeam, 
                { top: '0%', opacity: 0 },
                { 
                    top: '100%', 
                    opacity: 1, 
                    duration: 1.5, 
                    ease: "power1.inOut",
                    onUpdate: function() {
                        if (Math.random() > 0.6) createDetectionBox(this.progress() * 90);
                    }
                }
            );

            // Upward Scan (Analysis)
            tl.to(scannerBeam, {
                top: '0%',
                duration: 1.0,
                ease: "power2.out",
                onStart: () => statusText.innerText = 'Vectorizing...'
            });

            // Finish
            tl.to(scannerBeam, {
                opacity: 0,
                duration: 0.2,
                onComplete: () => {
                    scannerBeam.classList.add('hidden');
                    statusPill.classList.add('opacity-0', 'translate-y-4'); // Hide status
                    
                    // Show Result Sheet
                    resultSheet.classList.remove('translate-y-[110%]');
                    resultSheet.classList.add('translate-y-0');
                    
                    // Delay typing slightly for sheet animation
                    setTimeout(() => {
                        typeWriter(TARGET_TEXT, outputText);
                    }, 400);
                }
            });
        });

    </script>
